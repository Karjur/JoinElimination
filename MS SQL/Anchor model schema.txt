-- TYP_GenreType table
CREATE TABLE dbo.TYP_GenreType (
    TYP_ID INT NOT NULL,
    TYP_GenreType VARCHAR(25) NOT NULL,
    PRIMARY KEY (TYP_ID),
    CONSTRAINT uqTYP_GenreType UNIQUE (TYP_GenreType)
);

-- RTT_RatingType table
CREATE TABLE dbo.RTT_RatingType (
    RTT_ID SMALLINT NOT NULL,
    RTT_RatingType VARCHAR(25) NOT NULL,
    PRIMARY KEY (RTT_ID),
    CONSTRAINT uqRTT_RatingType UNIQUE (RTT_RatingType)
);

-- MV_Movie table
CREATE TABLE dbo.MV_Movie (
    MV_ID INT IDENTITY(1,1) PRIMARY KEY,
    MV_Dummy BIT
);

-- MV_NAM_Movie_Name table
CREATE TABLE dbo.MV_NAM_Movie_Name (
    MV_ID INT NOT NULL,
    MV_NAM_Movie_Name VARCHAR(50) NOT NULL,
    CONSTRAINT FK_MV_NAM_Movie_Name FOREIGN KEY (MV_ID) REFERENCES dbo.MV_Movie (MV_ID),
    PRIMARY KEY (MV_ID)
);

-- MV_GER_Movie_Gerne table
CREATE TABLE dbo.MV_GER_Movie_Gerne (
    MV_ID INT NOT NULL,
    TYP_ID INT NOT NULL,
    CONSTRAINT FK_A_MV_GER_Movie_Gerne FOREIGN KEY (MV_ID) REFERENCES dbo.MV_Movie (MV_ID),
    CONSTRAINT FK_K_MV_GER_Movie_Gerne FOREIGN KEY (TYP_ID) REFERENCES dbo.TYP_GenreType (TYP_ID),
    PRIMARY KEY (MV_ID)
);

-- MV_ATH_Movie_Author table
CREATE TABLE dbo.MV_ATH_Movie_Author (
    MV_ID INT NOT NULL,
    MV_ATH_Movie_Author VARCHAR(50) NOT NULL,
    CONSTRAINT FK_MV_ATH_Movie_Author FOREIGN KEY (MV_ID) REFERENCES dbo.MV_Movie (MV_ID),
    PRIMARY KEY (MV_ID)
);

-- MV_RAT_Movie_Rating table
CREATE TABLE dbo.MV_RAT_Movie_Rating (
    MV_ID INT NOT NULL,
    RTT_ID SMALLINT NOT NULL,
    MV_RAT_ChangedAt DATETIME2 NOT NULL,
    CONSTRAINT FK_A_MV_RAT_Movie_Rating FOREIGN KEY (MV_ID) REFERENCES dbo.MV_Movie (MV_ID),
    CONSTRAINT FK_K_MV_RAT_Movie_Rating FOREIGN KEY (RTT_ID) REFERENCES dbo.RTT_RatingType (RTT_ID),
    PRIMARY KEY (MV_ID, MV_RAT_ChangedAt)
);

-- Index for MV_RAT_Movie_Rating_MV_RAT_ChangedAt
CREATE INDEX IX_MV_RAT_Movie_Rating_MV_RAT_ChangedAt ON dbo.MV_RAT_Movie_Rating (MV_RAT_ChangedAt);

-- Views (latest perspective)
CREATE OR ALTER VIEW dbo.lMV_RAT_Movie_Rating AS
SELECT MV_ID, RTT_ID, MV_RAT_ChangedAt
FROM dbo.MV_RAT_Movie_Rating;

CREATE OR ALTER VIEW dbo.lMV_Movie AS
SELECT MV.MV_ID, NAM.MV_NAM_Movie_Name, kGER.TYP_GenreType AS TYP_GenreType,
    GER.TYP_ID, ATH.MV_ATH_Movie_Author, RAT.MV_RAT_ChangedAt,
    kRAT.RTT_RatingType AS RTT_RatingType, RAT.RTT_ID
FROM dbo.MV_Movie MV
LEFT JOIN dbo.MV_NAM_Movie_Name NAM ON NAM.MV_ID = MV.MV_ID
LEFT JOIN dbo.MV_GER_Movie_Gerne GER ON GER.MV_ID = MV.MV_ID
LEFT JOIN dbo.TYP_GenreType kGER ON kGER.TYP_ID = GER.TYP_ID
LEFT JOIN dbo.MV_ATH_Movie_Author ATH ON ATH.MV_ID = MV.MV_ID
LEFT JOIN dbo.lMV_RAT_Movie_Rating RAT ON RAT.MV_ID = MV.MV_ID
LEFT JOIN dbo.RTT_RatingType kRAT ON kRAT.RTT_ID = RAT.RTT_ID;
