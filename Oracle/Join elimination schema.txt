CREATE TABLE Valuuta (valuutakood CHAR ( 3 ) NOT NULL, 
nimetus VARCHAR2 ( 100 ) NOT NULL, 
CONSTRAINT PK_Valuuta PRIMARY KEY (valuutakood),
CONSTRAINT AK_Valuuta_nimetus UNIQUE (nimetus)
); 

CREATE TABLE Asukoht ( asukoht_ID NUMBER (5), 
valuutakood CHAR ( 3 ) NOT NULL, 
asukoht VARCHAR2 ( 255 ) NOT NULL, 
CONSTRAINT PK_Asukoht PRIMARY KEY (asukoht_ID),
CONSTRAINT AK_Asukoht UNIQUE (asukoht)); 

CREATE TABLE Osakond ( osakond_ID NUMBER (5), 
asukoht_ID NUMBER ( 5 ) NOT NULL, 
nimi VARCHAR2 ( 100 ) NOT NULL, 
CONSTRAINT PK_Osakond PRIMARY KEY (Osakond_ID),
CONSTRAINT AK_Osakond UNIQUE (nimi));

CREATE TABLE Tootaja (tootaja_ID NUMBER (10) GENERATED BY DEFAULT AS IDENTITY, 
osakond_ID NUMBER ( 5 ) NOT NULL, 
eesnimi VARCHAR2 ( 50 ) NOT NULL, 
palk DECIMAL(10,2) NOT NULL, 
perenimi VARCHAR2 ( 50 ), 
alates DATE NOT NULL, 
CONSTRAINT PK_Tootaja PRIMARY KEY (tootaja_ID) ); 

CREATE TABLE Projekt ( projekt_ID NUMBER (10) GENERATED BY DEFAULT AS IDENTITY, 
tootaja_ID NUMBER ( 10 ) NOT NULL, 
tahtaeg DATE, 
prioriteet NUMBER(3), 
CONSTRAINT PK_projekt PRIMARY KEY (projekt_ID) ); 

ALTER TABLE Asukoht ADD ( CONSTRAINT FK_Asukoht_Valuuta FOREIGN KEY (valuutakood) REFERENCES Valuuta (valuutakood)); 
ALTER TABLE Osakond ADD ( CONSTRAINT FK_Osakond_Asukoht FOREIGN KEY (asukoht_ID) REFERENCES Asukoht (asukoht_ID)); 
ALTER TABLE Tootaja ADD ( CONSTRAINT FK_Tootaja_Osakond FOREIGN KEY (osakond_ID) REFERENCES Osakond (osakond_ID)); 

CREATE INDEX idx_asukoht_valuuta ON Asukoht(valuutakood);
CREATE INDEX idx_osak_asukoht ON Osakond(asukoht_ID); 
CREATE INDEX idx_tootaja_osak ON Tootaja(osakond_ID); 

CREATE OR REPLACE VIEW Tootaja_osakond AS 
SELECT Tootaja.perenimi AS tootaja, Osakond.nimi AS osakond 
FROM Tootaja INNER JOIN Osakond ON Tootaja.osakond_id=Osakond.osakond_id; 

CREATE OR REPLACE VIEW Tootaja_osakond_klassikaline AS 
SELECT Tootaja.perenimi, Osakond.nimi AS osakond 
FROM Tootaja, Osakond 
WHERE Tootaja.osakond_ID=Osakond.osakond_ID;

DROP VIEW Tootaja_osakond;
DROP VIEW Tootaja_osakond_klassikaline;
DROP TABLE Projekt;
DROP TABLE Tootaja;
DROP TABLE Osakond;
DROP TABLE Asukoht;
DROP TABLE Valuuta;
